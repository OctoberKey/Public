105 cv::Vec3d R2Angles(const cv::Matx33d& R)
106 {
107  cv::Vec3d angles;
108  angles[0] = atan2(R(1, 2), R(2, 2));
109  angles[1] = atan2(-R(0, 2), sqrt(pow(R(1, 2), 2) + pow(R(2, 2), 2)));
110  angles[2] = atan2(R(0, 1), R(0, 0));
111  angles[0] = Rad2Deg(angles[0]);
112  angles[1] = Rad2Deg(angles[1]);
113  angles[2] = Rad2Deg(angles[2]);
114  return angles;
115 }
